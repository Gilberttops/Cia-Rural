Atualize o Apps Script para aceitar lançamentos

Substitua o código do script por este (mantém a leitura e adiciona a gravação). Troque SUA-CHAVE-AQUI por uma senha sua — ela impede que qualquer pessoa com o link grave na planilha. Depois, Implantar → Gerenciar implantações → editar → Nova versão (a URL /exec continua a mesma). 
const CHAVE = 'SUA-CHAVE-AQUI'; // deixe '' para desativar a proteção

function doGet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const out = {};
  out['_url'] = ss.getUrl();
  ['Cargas', 'Baixas ', 'Baixas', 'Chacobras'].forEach(function (nome) {
    const aba = ss.getSheetByName(nome);
    if (aba && !out[nome.trim()]) {
      out[nome.trim()] = aba.getDataRange().getDisplayValues();
    }
  });
  return resposta(out);
}

function doPost(e) {
  try {
    const d = JSON.parse(e.postData.contents);
    if (CHAVE && d.chave !== CHAVE) return resposta({ ok: false, erro: 'chave inválida' });
    const ss = SpreadsheetApp.getActiveSpreadsheet();

    if (d.tipo === 'venda') {
      const aba = ss.getSheetByName(d.empresa === 'Chacobras' ? 'Chacobras' : 'Cargas');
      const data = new Date(d.data + 'T12:00:00');
      const venc = d.venc ? new Date(d.venc + 'T12:00:00') : '';
      let subtotal = 0;
      d.itens.forEach(function (it) { subtotal += it.qtd * it.unit; });
      const valores = d.itens.map(function (it, i) {
        return [data, it.produto, it.qtd, it.unit, it.qtd * it.unit,
                i === 0 ? subtotal : '', i === 0 ? venc : ''];
      });
      inserirAposUltimaData(aba, valores, 7);
      return resposta({ ok: true, linhas: d.itens.length });
    }

    if (d.tipo === 'baixa') {
      const aba = ss.getSheetByName('Baixas ') || ss.getSheetByName('Baixas');
      const valores = [[new Date(d.data + 'T12:00:00'), d.valor_brl || '', d.cotacao || '',
                        d.valor_usd, d.obs || '']];
      inserirAposUltimaData(aba, valores, 5);
      return resposta({ ok: true });
    }

    return resposta({ ok: false, erro: 'tipo desconhecido' });
  } catch (err) {
    return resposta({ ok: false, erro: String(err) });
  }
}

/* insere logo abaixo da última linha que tem DATA na coluna A,
   empurrando o bloco de totais para baixo */
function inserirAposUltimaData(aba, valores, numCols) {
  const colA = aba.getRange(1, 1, aba.getLastRow(), 1).getValues();
  let ultima = 0;
  for (let i = 0; i < colA.length; i++) {
    if (colA[i][0] instanceof Date) ultima = i + 1;
  }
  if (ultima === 0) { valores.forEach(function (v) { aba.appendRow(v); }); return; }
  aba.insertRowsAfter(ultima, valores.length);
  aba.getRange(ultima + 1, 1, valores.length, numCols).setValues(valores);
}

function resposta(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

Sincronizar com o Google Sheets

O dashboard busca os dados toda vez que você abre ou clica em atualizar.

Código do Apps Script (cole no editor de script da planilha)

No Google Sheets: Extensões → Apps Script, cole o código abaixo, depois Implantar → Nova implantação → App da web (executar como você, acesso: qualquer pessoa com o link) e cole a URL /exec no campo acima. Ele devolve as três abas de uma vez, no formato exibido na planilha — que é o que o dashboard entende. 
function doGet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const out = {};
  out['_url'] = ss.getUrl();
  ['Cargas', 'Baixas ', 'Baixas', 'Chacobras'].forEach(function (nome) {
    const aba = ss.getSheetByName(nome);
    if (aba && !out[nome.trim()]) {
      out[nome.trim()] = aba.getDataRange().getDisplayValues();
    }
  });
  return ContentService
    .createTextOutput(JSON.stringify(out))
    .setMimeType(ContentService.MimeType.JSON);
}
Importante: use getDisplayValues() (como acima), que envia datas e números do jeito que aparecem na planilha. A cada alteração no código do script é preciso reimplantar. A URL /exec dá acesso aos dados a quem a tiver — mantenha-a privada.

Alternativa CSV: se preferir não usar script, mude para "CSV publicado" acima e use Arquivo → Compartilhar → Publicar na web → formato CSV, uma aba por link (Cargas, Baixas e Chacobras).
Apenas as abas visíveis da planilha: Cargas (Ypoti) e Chacobras para as vendas, e Baixas para os pagamentos. Valores em dólares (US$).
As ligações entre baixas e vendas são feitas por você nesta ferramenta e ficam salvas no navegador — não alteram a planilha original.
